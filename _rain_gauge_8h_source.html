114typedef struct {
115 /* Timestamp of last update */
116 time_t lastUpdate;
117
118 /* Data of past 60 minutes */
119 int16_t hist[RAIN_HIST_SIZE];
120
121 /* Sensor startup handling */
122 bool startupPrev; // previous state of startup
123 float rainPreStartup; // previous rain gauge reading (before startup)
124
125 /* Rainfall of current day (can start anytime, but will reset on begin of new day) */
126 uint8_t tsDayBegin; // day of week
127 float rainDayBegin; // rain gauge @ begin of day
128
129 /* Rainfall of current week (can start anytime, but will reset on Monday */
130 uint8_t tsWeekBegin; // day of week
131 float rainWeekBegin; // rain gauge @ begin of week
132 uint8_t wdayPrev; // day of week at previous run - to detect new week
133
134 /* Rainfall of current calendar month (can start anytime, but will reset at begin of month */
135 uint8_t tsMonthBegin; // month
136 float rainMonthBegin; // rain gauge @ begin of month
137
138 float rainPrev; // rain gauge at previous run - to detect overflow
139 float rainAcc; // accumulated rain (overflows and startups)
140} nvData_t;
