128typedef struct {

138 /* Sensor startup handling */

142 /* Rainfall of current day (can start anytime, but will reset on begin of new day) */
143 uint8_t tsDayBegin; // day of week
144 float rainDayBegin; // rain gauge @ begin of day
145
146 /* Rainfall of current week (can start anytime, but will reset on Monday */
147 uint8_t tsWeekBegin; // day of week

149 uint8_t wdayPrev; // day of week at previous run - to detect new week
150
151 /* Rainfall of current calendar month (can start anytime, but will reset at begin of month */
152 uint8_t tsMonthBegin; // month

154
155 float rainPrev; // rain gauge at previous run - to detect overflow
156 float rainAcc; // accumulated rain (overflows and startups)
157
158 uint8_t updateRate; // update rate for pastHour() calculation
159} nvData_t;
