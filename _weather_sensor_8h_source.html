103// Sensor Types
104// 0 - Weather Station 5-in-1; PN 7002510..12/7902510..12
105// 1 - Weather Station 6-in-1; PN 7002585
106// - Professional Wind Gauge 6-in-1; PN 7002531
107// - Weather Station 7-in-1; PN 7003300
108// 2 - Thermo-/Hygro-Sensor 6-in-1; PN 7009999
109// 3 - Pool / Spa Thermometer 6-in-1; PN 7000073
110// 4 - Soil Moisture Sensor 6-in-1; PN 7009972
111// 5 - Water Leakage Sensor 6-in-1; PN 7009975
112// 8 - Air Quality Sensor PM2.5/PM10 7-in-1; P/N 7009970
113// 9 - Professional Rain Gauge (5-in-1 decoder)
114// 9 - Lightning Sensor PN 7009976
115// 10 - CO2 Sensor 7-in-1; PN 7009977
116// 11 - HCHO/VCO Sensor 7-in-1; PN 7009978
117#define SENSOR_TYPE_WEATHER0 0 // Weather Station
118#define SENSOR_TYPE_WEATHER1 1 // Weather Station
119#define SENSOR_TYPE_THERMO_HYGRO 2 // Thermo-/Hygro-Sensor
120#define SENSOR_TYPE_POOL_THERMO 3 // Pool / Spa Thermometer
121#define SENSOR_TYPE_SOIL 4 // Soil Temperature and Moisture (from 6-in-1 decoder)
122#define SENSOR_TYPE_LEAKAGE 5 // Water Leakage
123#define SENSOR_TYPE_AIR_PM 8 // Air Quality Sensor (Particle Matter)
124#define SENSOR_TYPE_RAIN 9 // Professional Rain Gauge (from 5-in-1 decoder)
125#define SENSOR_TYPE_LIGHTNING 9 // Lightning Sensor
126#define SENSOR_TYPE_CO2 10 // CO2 Sensor
127#define SENSOR_TYPE_HCHO_VOC 11 // Air Quality Sensor (HCHO and VOC)

355 void clearSlots(uint8_t type = 0xFF)
356 {
357 for (size_t i=0; i<sensor.size(); i++) {
358 if ((type == 0xFF) || (sensor[i].s_type == type)) {
359 sensor[i].valid = false;
360 sensor[i].complete = false;
361 }
362 if (sensor[i].s_type == SENSOR_TYPE_WEATHER1) {
363 sensor[i].w.temp_ok = false;
364 sensor[i].w.humidity_ok = false;
365 sensor[i].w.light_ok = false;
366 sensor[i].w.uv_ok = false;
367 sensor[i].w.wind_ok = false;
368 sensor[i].w.rain_ok = false;
369 }
370 }

632 const char txt[] = "Byte #: ";
633 int offs;
634 int len1 = strlen(txt);

636 int prefix_len = max(len1, len2);
637

640 offs = (len1 < len2) ? (len2 - len1) : 0;
641 strcpy(&buf[offs], txt);
642
643 // Print byte index
644 for (size_t i = 0 ; i < msgSize; i++) {
645 sprintf(&buf[strlen(buf)], "%02d ", i);

648

650 buf[prefix_len] ='\0';
651 offs = (len1 > len2) ? (len1 - len2) : 0;
652 sprintf(&buf[offs], "%s: ", descr);
653
654 for (size_t i = 0 ; i < msgSize; i++) {
